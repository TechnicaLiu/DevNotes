1. 求数组1-N中出现1的次数

   ```js
   // 思路一 : 将数组转换成字符串再转换成数组
    function countTotal(arr) {
    let count = 0;
       let res = arr.join('').split('');
       res.forEach(item => {
         if (item == 1) {
           count++
         }
       })
    }
   // 思路二 : 数组转换成字符串，利用正则表达式  字符串.match(正则) 成功返回数组，失败返回null
     function regStr(arr) {
         return arr.reduce((pre, cur) => {
           const curCount = String(cur).match(/1/g) ? .length || 0;
           return pre + curCount
         }, 0)
       }
   ```
